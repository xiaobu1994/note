```java
package com.xiaobu.util;

import java.io.File;

/**
 * @author xiaobu
 * @version JDK1.8.0_171
 * @date on  2019/4/6 22:38
 * @description V1.0 遍历获取文件
 */
public class Traverse2 {


    public static void main(String[] args) {
        String path = "C:\\Users\\Administrator\\Desktop\\4-4";
        File file = new File(path);
        getFiles(file);
    }

    private static void getFiles(File file) {
        File[] fs = file.listFiles();
        assert fs != null;
        for (File f : fs) {
            if (f.isDirectory()) {
                System.out.println("文件夹:"+f);
                getFiles(f);
            }
            if (f.isFile()) {
                System.out.println("文件："+f);
            }
        }
    }


}
```
