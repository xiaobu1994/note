引入依赖

```xml
<dependency>
            <groupId>org.apache.commons</groupId>
            <artifactId>commons-lang3</artifactId>
            <version>3.4</version>
        </dependency>
        <dependency>
            <groupId>commons-lang</groupId>
            <artifactId>commons-lang</artifactId>
            <version>2.6</version>
        </dependency>
```

代码

```java
package com.xiaobu.note.StringDemo;

import org.apache.commons.lang.RandomStringUtils;

/**
 * @author xiaobu
 * @version JDK1.8.0_171
 * @date on  2019/11/25 16:20
 * @description
 */
public class RandomStringDemo {
    public static void main(String[] args) {
        commonsLang3RandomString();
        commonsLangRandomString();
    }


    public static void commonsLangRandomString(){
        String result = RandomStringUtils.random(64, true, false);
        System.out.println("result = " + result);
    }

    public static void commonsLang3RandomString(){
        String result = org.apache.commons.lang3.RandomStringUtils.random(64, true, false);
        System.out.println("result = " + result);
    }

}
```
