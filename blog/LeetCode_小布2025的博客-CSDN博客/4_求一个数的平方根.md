package com.xiaobu.leetcode;

/**
 * @author xiaobu
 * @version JDK1.8.0_171
 * @date on  2019/10/23 16:51
 * @description 求一个数的平方根
 */
public class GetTheSqrtOfNumber {

    public static void main(String[] args) {
      int result=  getMySqrt(99);
        System.out.println("result = " + result);
    }


    /**
     * 功能描述:二分法
     * @author xiaobu
     * @date 2020/5/7 17:35
     * @param x int
     * @return int
     * @version 1.0
     */
    public static int getMySqrt(int x){
        int max=x;
        int min=0;
        while (max-min>1){
            int m = (max + min) / 2;
            if(x/m<m){
                max = m;
            }else {
                min = m;
            }
        }
        return min;
    }



    public static int GetTheSqrtOfNumberByMath(double number) {
        return (int) Math.sqrt(number);
    }
}
