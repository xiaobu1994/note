package com.xiaobu.base.utils;

import java.text.ParseException;
import java.text.SimpleDateFormat;
import java.util.Date;

/**
 * 时间戳转日期
 * @author xiaobu
 * @version JDK1.8.0_171
 * @date on  2019/4/4 10:19
 * @description V1.0
 */
public class TimeStampUtil {

    public static void main(String[] args) {
        timeStamp2Date(1554619113000L,"yyyy-MM-dd HH:mm:ss");
    }



    /**
     * 功能描述:日期转时间戳
     * @author xiaobu
     * @date 2019/4/4 10:44
     * @return void
     * @version 1.0
     */
    public static  void date2TimeStamp(String dateStr,String pattern){
        SimpleDateFormat simpleDateFormat = new SimpleDateFormat(pattern);
        try {
            Date date = simpleDateFormat.parse(dateStr);
            System.out.println("date.getTime() = " + date.getTime());
        } catch (ParseException e) {
            e.printStackTrace();
        }
    }


    /**
     * 功能描述:时间戳转日期 若为10位则需要乘以1000 若是13位这不需要
     * @author xiaobu
     * @date 2019/4/4 10:45
     * @param timeStamp 时间戳
     * @return void  yyyy-MM-dd HH:mm:ss
     * @version 1.0
     */
    public static void timeStamp2Date(long timeStamp,String pattern){
        SimpleDateFormat simpleDateFormat = new SimpleDateFormat(pattern);
        Date date = new Date(timeStamp);
        String s=simpleDateFormat.format(date);
        System.out.println("s = " + s);
    }

    
    
    



}
